        width: 100px;
        height: 100px;
        background-color: chartreuse;
        margin: 0 auto;
      }
  
      .demo2::before {
        content: '';
        width:100px;
        height: 100px;
        display: inline-block;
        background-color:#fff;
        transform:translateX(-50%) skew(-45deg);
        /* 代表的是扭曲 */
      }

第三种方法：H5画布制作三角形 
      
      var oCanvas = document.getElementById('oCanvas')
      var ctx = oCanvas.getContext('2d')
      ctx.beginPath();
      ctx.moveTo(50,0);
      ctx.lineTo(0,50);
      ctx.lineTo(100,50);
      ctx.closePath();
      // 闭合路径
      ctx.stroke();
      // 描边
      ctx.fillStyle = '#f00'
      ctx.fill()
